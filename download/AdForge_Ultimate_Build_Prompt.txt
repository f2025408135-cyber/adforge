SECTION 1: PROJECT OVERVIEW
================================================================================

AdForge is an AI-powered advertisement campaign generator that takes a product
brief as input and produces complete, ready-to-publish ad campaigns including
headlines, taglines, ad copy, calls to action, target audience profiles, key
benefits, and platform-specific adaptations. It supports multiple AI providers,
has a professional editorial-style design system, and must feel like a premium
SaaS product — not a toy or a hackathon project.

The current prototype exists at https://thinkaiproject.netlify.app but needs a
complete rebuild from the ground up with professional architecture, a proper
backend, database persistence, authentication, real-time features, and a much
richer feature set.

SECTION 4: DATABASE SCHEMA (PRISMA — SQLITE)
================================================================================

Replace the existing prisma/schema.prisma with this complete schema:

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

model User {
  id            String    @id @default(cuid())
  email         String    @unique
  name          String?
  image         String?
  passwordHash  String?   // for credentials auth
  provider      String    @default("credentials") // credentials | google | github
  role          String    @default("user") // user | admin
  plan          String    @default("free") // free | pro | enterprise
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  campaigns     Campaign[]
  brandKits     BrandKit[]
  teamMemberships TeamMember[]
  apiUsage      ApiUsage[]
}

model Campaign {
  id            String    @id @default(cuid())
  userId        String
  productName   String
  productDesc   String
  tone          String    @default("professional")
  audience      String?
  platforms     String    @default("instagram,facebook") // comma-separated
  provider      String    @default("gemini")
  headline      String?
  tagline       String?
  adCopy        String?
  callToAction  String?
  targetAudience String?
  keyBenefits   String?
  platformVersions String?
  status        String    @default("completed") // draft | completed | archived
  isFavorite    Boolean   @default(false)
  variants      Json?     // stores A/B test variants as JSON array
  tags          String?   // comma-separated tags
  rating        Int?      // 1-5 star rating
  notes         String?   // user notes/feedback
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  exports       Export[]
  versions      CampaignVersion[]

  @@index([userId])
  @@index([createdAt])
}

model CampaignVersion {
  id            String    @id @default(cuid())
  campaignId    String
  version       Int
  headline      String?
  tagline       String?
  adCopy        String?
  callToAction  String?
  targetAudience String?
  keyBenefits   String?
  platformVersions String?
  changeSummary String?   // e.g. "Regenerated headline"
  createdAt     DateTime  @default(now())

  campaign      Campaign  @relation(fields: [campaignId], references: [id], onDelete: Cascade)

  @@index([campaignId])
}

model BrandKit {
  id            String    @id @default(cuid())
  userId        String
  name          String
  brandName     String
  brandVoice    String    // brand tone/voice description
  primaryColor  String?
  secondaryColor String?
  logo          String?   // URL or base64
  guidelines    String?   // brand guidelines text
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
}

model TeamMember {
  id            String    @id @default(cuid())
  userId        String
  teamId        String    // for future team/org features
  role          String    @default("viewer") // owner | editor | viewer
  invitedAt     DateTime  @default(now())
  acceptedAt    DateTime?

  user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)
}

model Export {
  id            String    @id @default(cuid())
  campaignId    String
  format        String    // pdf | docx | markdown | txt | png | json
  filePath      String?
  createdAt     DateTime  @default(now())

  campaign      Campaign  @relation(fields: [campaignId], references: [id], onDelete: Cascade)

  @@index([campaignId])
}

model ApiUsage {
  id            String    @id @default(cuid())
  userId        String
  provider      String
  endpoint      String    // generate | regenerate | analyze
  tokensUsed    Int       @default(0)
  createdAt     DateTime  @default(now())

  user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([createdAt])
}

model Template {
  id            String    @id @default(cuid())
  name          String
  description   String
  category      String    // product | service | event | saas | ecommerce
  promptTemplate String   // the system prompt with {{variables}}
  tone          String    @default("professional")
  isPublic      Boolean   @default(true)
  usageCount    Int       @default(0)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
}
```

After writing the schema, run: npx prisma db push

------------------------------------------------------------------------
8.1 POST /api/generate
------------------------------------------------------------------------

Request Body:
{
  provider: "gemini" | "deepseek" | "glm",
  productName: string,
  productDesc: string,
  tone: string,
  audience?: string,
  platforms: string[],
  brandVoice?: string,       // from brand kit
  language?: string,          // "en" default
  creativity?: number,        // 0-100, default 50
  templateId?: string,        // optional template
  additionalInstructions?: string
}

Process:
1. Validate required fields
2. Get server-side API key for the provider
3. If templateId provided, load template prompt from database
4. Build the system prompt incorporating:
   - Product details
   - Tone (mapped to full description)
   - Platform-specific requirements
   - Brand voice (if provided)
   - Language instruction (if not English)
   - Template structure (if provided)
   - Additional instructions (if provided)
5. Map creativity (0-100) to temperature:
   - 0-33 → temperature 0.3 (conservative)
   - 34-66 → temperature 0.7 (balanced)
   - 67-100 → temperature 1.0 (creative)
6. Call the selected AI provider API
7. Parse the JSON response into structured sections
8. Return the result

Response:
{
  result: {
    headline: string,
    tagline: string,
    adCopy: string,
    callToAction: string,
    targetAudience: string,
    keyBenefits: string,
    platformVersions: string
  },
  provider: string,
  model: string,
  tokensUsed: number
}
